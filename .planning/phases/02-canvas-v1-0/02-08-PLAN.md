               'infracanvas.cost.estimator',
               'infracanvas.drift.analyzer',
               'infracanvas.export.html',
               'infracanvas.export.json',

               'infracanvas.config',
               'hcl2',
               'yaml',

               'typer',
               'rich',

           hookspath=[],

           upx_exclude=[],

           disable_windowed_traceback=False,

    Create `.github/workflows/release.yml`:
    ```yaml
    name: Release

    on:
      push:
        tags:
          - 'v*'

    permissions:
      contents: write
      packages: write

    jobs:
      build-binaries:
        name: Build ${{ matrix.os }} binary
        runs-on: ${{ matrix.os }}
        strategy:
          matrix:
            include:
              - os: ubuntu-latest
                artifact: infracanvas-linux-amd64
                ext: ""
              - os: macos-14
                artifact: infracanvas-macos-arm64
                ext: ""
              - os: windows-latest
                artifact: infracanvas-windows-x64
                ext: ".exe"

        steps:
          - uses: actions/checkout@v4

          - uses: actions/setup-python@v5
            with:
              python-version: '3.12'

          - name: Install dependencies
            run: |
              cd cli
              pip install -e .
              pip install pyinstaller

          - name: Build viewer (for HTML template)
            run: |
              cd viewer
              npm ci
              npm run build

          - name: Build binary
            run: |
              cd cli
              pyinstaller infracanvas.spec

          - name: Rename artifact
            shell: bash
            run: |
              mv cli/dist/infracanvas${{ matrix.ext }} cli/dist/${{ matrix.artifact }}${{ matrix.ext }}

          - uses: actions/upload-artifact@v4
            with:
              name: ${{ matrix.artifact }}
              path: cli/dist/${{ matrix.artifact }}${{ matrix.ext }}

      build-docker:
        name: Build Docker image
        runs-on: ubuntu-latest
        needs: []
        steps:
          - uses: actions/checkout@v4

          - uses: actions/setup-python@v5
            with:
              python-version: '3.12'

          - name: Build viewer
            run: |
              cd viewer
              npm ci
              npm run build

          - uses: docker/setup-buildx-action@v3

          - uses: docker/login-action@v3
            with:
              registry: ghcr.io
              username: ${{ github.actor }}
              password: ${{ secrets.GITHUB_TOKEN }}

          - name: Build and push Docker image
            uses: docker/build-push-action@v6
            with:
              context: .
              push: true
              platforms: linux/amd64,linux/arm64
              tags: |
                ghcr.io/${{ github.repository }}:${{ github.ref_name }}
                ghcr.io/${{ github.repository }}:latest

      publish-pypi:
        name: Publish to PyPI
        runs-on: ubuntu-latest
        needs: []
        steps:
          - uses: actions/checkout@v4

          - uses: actions/setup-python@v5
            with:
              python-version: '3.12'

          - name: Build package
            run: |
              cd cli
              pip install build
              python -m build

          - name: Publish to PyPI
            uses: pypa/gh-action-pypi-publish@release/v1
            with:
              packages-dir: cli/dist/

      create-release:
        name: Create GitHub Release
        runs-on: ubuntu-latest
        needs: [build-binaries, build-docker, publish-pypi]
        steps:
          - uses: actions/download-artifact@v4
            with:
              path: artifacts/

          - name: Create release
            uses: softprops/action-gh-release@v2
            with:
              files: |
                artifacts/infracanvas-linux-amd64/infracanvas-linux-amd64
                artifacts/infracanvas-macos-arm64/infracanvas-macos-arm64
                artifacts/infracanvas-windows-x64/infracanvas-windows-x64.exe
              generate_release_notes: true
    ```
  </action>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions -> PyPI | Package published to public registry |
| GitHub Actions -> GHCR | Docker image pushed to container registry |
| Dockerfile -> runtime | Container executes user-provided Terraform scans |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-16 | Tampering | release.yml PyPI publish | mitigate | Uses official pypa/gh-action-pypi-publish action. PyPI API token stored as GitHub secret. |
| T-02-17 | Elevation of Privilege | Dockerfile | mitigate | Non-root user (infracanvas). No SUID binaries. Read-only scanning — no write operations on host. |
| T-02-18 | Information Disclosure | Docker image layers | accept | Image contains open-source code (MIT license). No secrets in image layers. |
</threat_model>
