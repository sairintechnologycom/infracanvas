```python
        # ---- 9. Best-effort Slack alert for webhook-triggered scans. -----
        # Fires only if: source='webhook', slack_webhook_url set, >=1 Critical.
        # Never re-raises — Slack failure must not flip a completed scan to failed.
        # Critical count key path: summary_json["findings"]["critical"]
        # (_extract_summary returns the GraphSummary dict; GraphSummary.findings
        #  is a dict[str, int] with key "critical" — confirmed from models.py line 69)
        try:
            sm2 = get_sessionmaker()
            async with sm2() as _session:
                async with _session.begin():
                    _row = (
                        await _session.execute(
                            text(
                                "SELECT s.source, t.slack_webhook_url "
                                "FROM scans s "
                                "JOIN teams t ON t.id = s.team_id "
                                "WHERE s.id = :sid"
                            ),
                            {"sid": scan_id},
                        )
                    ).one_or_none()
            if _row is not None:
                _source, _slack_url = _row.source, _row.slack_webhook_url
                _critical = (summary_json or {}).get("findings", {}).get("critical", 0)
                if _source == "webhook" and _slack_url and _critical >= 1:
                    _msg = {
                        "text": f"⚠️ {_critical} Critical finding(s) on {repo}@{sha[:7]}",
                        "blocks": [
                            {
                                "type": "section",
                                "text": {
                                    "type": "mrkdwn",
                                    "text": (
                                        f"*InfraCanvas Critical Alert*\n"
                                        f"Repo: `{repo}`\n"
                                        f"Branch: `{branch}`\n"
                                        f"Commit: `{sha[:7]}`\n"
                                        f"Critical findings: {_critical}\n"
                                        f"<{settings.dashboard_url}/scans/{scan_id}|View scan>"
                                    ),
                                },
                            }
                        ],
                    }
                    async with httpx.AsyncClient() as _http:
                        await _http.post(_slack_url, json=_msg, timeout=5.0)
                    log_ctx.info("slack_alert_sent", slack_url=_slack_url[:40])
        except Exception as _slack_exc:
            log_ctx.warning("slack_fire_failed", error=str(_slack_exc))
            sentry_sdk.capture_exception(_slack_exc)
```

Note: `summary_json` is already in scope at this point (set at line ~249 via `_extract_summary(payload_bytes)`). The key path `summary_json.get("findings", {}).get("critical", 0)` is confirmed correct: `_extract_summary` returns the `summary` dict from the scan JSON, which is the serialized `GraphSummary` model. `GraphSummary.findings` is a `dict[str, int]` with key `"critical"` (cli/infracanvas/graph/models.py line 69). `settings` needs to be imported — add `from app.config import settings` to the imports if not already present (grep first). `repo`, `branch`, `sha`, `scan_id`, `team_id` are all in scope from the function parameters.
