    ```python
    from __future__ import annotations
    import logging, re, sys
    from typing import Any
    import orjson, structlog

    _SCRUB_KEYS = {"authorization", "x-signature", "cookie", "stripe-signature",
                   "svix-signature", "svix-id", "clerk-webhook-secret",
                   "stripe_secret_key", "r2_secret_access_key", "clerk_webhook_secret"}
    _R2_URL_RE = re.compile(r"(https://[^.]+\.r2\.cloudflarestorage\.com/[^?\s]+)\?[^\s\"]+")

    def scrub_sensitive(_, __, event_dict: dict[str, Any]) -> dict[str, Any]:
        for k in list(event_dict.keys()):
            if k.lower() in _SCRUB_KEYS:
                event_dict[k] = "<redacted>"
        for k, v in event_dict.items():
            if isinstance(v, str):
                event_dict[k] = _R2_URL_RE.sub(r"\1?<redacted>", v)
        return event_dict

    def configure_logging(*, level: int = logging.INFO) -> None:
        structlog.configure(
            processors=[
                structlog.contextvars.merge_contextvars,
                structlog.stdlib.add_log_level,
                structlog.processors.TimeStamper(fmt="iso"),
                scrub_sensitive,
                structlog.processors.StackInfoRenderer(),
                structlog.processors.dict_tracebacks,
                structlog.processors.JSONRenderer(serializer=orjson.dumps),
            ],
            wrapper_class=structlog.make_filtering_bound_logger(level),
            logger_factory=structlog.PrintLoggerFactory(file=sys.stdout),
            cache_logger_on_first_use=True,
        )
    ```

    **`backend/app/obs/middleware.py`** — PURE ASGI, not BaseHTTPMiddleware (RESEARCH § F9 lines 652-677, § P1 lines 1141-1144). Copy the code sketch verbatim with the access-log addition:
    ```python
    from __future__ import annotations
    import time
    from typing import Any
    import structlog
    from app.util.ids import new_uuid7

    _log = structlog.get_logger("app.http")

    class RequestContextMiddleware:
        """Pure ASGI middleware — DO NOT convert to BaseHTTPMiddleware.
        See RESEARCH.md § P1 / research callout #4: BaseHTTPMiddleware copies
        context; contextvars set inside would be invisible to the handler."""
        def __init__(self, app: Any) -> None:
            self.app = app

        async def __call__(self, scope: dict, receive: Any, send: Any) -> None:
            if scope["type"] != "http":
                await self.app(scope, receive, send)
                return

            hdrs = {k.decode().lower(): v.decode() for k, v in scope.get("headers", [])}
            rid = hdrs.get("x-request-id") or str(new_uuid7())
            structlog.contextvars.clear_contextvars()
            structlog.contextvars.bind_contextvars(request_id=rid)

            start = time.perf_counter()
            status_holder = {"status": 0}

            async def send_wrapper(msg: dict) -> None:
                if msg["type"] == "http.response.start":
                    status_holder["status"] = msg.get("status", 0)
                    headers = list(msg.get("headers", []))
                    headers.append((b"x-request-id", rid.encode()))
                    msg["headers"] = headers
                await send(msg)

            try:
                await self.app(scope, receive, send_wrapper)
            finally:
                duration_ms = round((time.perf_counter() - start) * 1000, 2)
                _log.info(
                    "request",
                    method=scope.get("method"),
                    path=scope.get("path"),
                    status=status_holder["status"],
                    duration_ms=duration_ms,
                )
                structlog.contextvars.clear_contextvars()
    ```

    from __future__ import annotations
    import json, io, sys
    from fastapi.testclient import TestClient
    from app.main import create_app

    def test_request_log_is_json_with_required_fields(capsys):
        """OBS-001: every request produces one JSON log line with request_id + method + path + status + duration_ms."""
        with TestClient(create_app()) as client:
            rid = "test-obs-01"
            client.get("/healthz", headers={"X-Request-ID": rid})
        out = capsys.readouterr().out.strip().splitlines()
        request_lines = []
        for line in out:
            try:
                payload = json.loads(line)
            except json.JSONDecodeError:
                continue
            if payload.get("event") == "request":
                request_lines.append(payload)
        assert len(request_lines) >= 1, f"no 'request' log lines found in {out!r}"
        entry = request_lines[-1]
        assert entry["request_id"] == rid
        assert entry["method"] == "GET"
        assert entry["path"] == "/healthz"
        assert entry["status"] == 200
        assert isinstance(entry["duration_ms"], (int, float))
