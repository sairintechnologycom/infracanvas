    ```python
    """Validate invariants across backend/fly.*.toml files.

    Run: python -m scripts.validate_fly_toml fly.dev.toml fly.prod.toml
    Exit 0 if all invariants hold, exit 1 otherwise.
    """
    from __future__ import annotations
    import sys
    from pathlib import Path
    try:
        import tomllib
    except ImportError:
        import tomli as tomllib  # py<3.11

    INVARIANTS = [
        ("[processes] must include 'api'",
            lambda d: "api" in d.get("processes", {})),
        ("[processes] must include 'worker'",
            lambda d: "worker" in d.get("processes", {})),
        ("release_command must run alembic upgrade head",
            lambda d: "alembic" in (d.get("deploy", {}).get("release_command", "")
                                     or "") and "upgrade head" in d["deploy"]["release_command"]),
        ("release_command_timeout must be >= 10m",
            lambda d: (d.get("deploy", {}).get("release_command_timeout", "0m")
                       or "0m").rstrip("m").isdigit()
                      and int(d["deploy"]["release_command_timeout"].rstrip("m")) >= 10),
        ("http_service.processes must be ['api'] (worker not exposed)",
            lambda d: d.get("http_service", {}).get("processes") == ["api"]),
        ("http_service.force_https must be true",
            lambda d: d.get("http_service", {}).get("force_https") is True),
        ("http_service.min_machines_running must be >= 1",
            lambda d: d.get("http_service", {}).get("min_machines_running", 0) >= 1),
        ("healthz http_check must exist",
            lambda d: any(c.get("path") == "/healthz"
                          for c in d.get("services", {}).get("http_checks", []))
                      or any(c.get("path") == "/healthz"
                             for c in d.get("services.http_checks", []) or [])),
    ]

    def validate(path: Path) -> list[str]:
        doc = tomllib.loads(path.read_text())
        # flatten nested [[services.http_checks]] (TOML's treatment differs per parser)
        failures: list[str] = []
        for desc, check in INVARIANTS:
            try:
                if not check(doc):
                    failures.append(f"{path.name}: {desc}")
            except Exception as e:
                failures.append(f"{path.name}: {desc} (raised {e!r})")
        return failures

    def main(argv: list[str]) -> int:
        failures: list[str] = []
        for arg in argv[1:]:
            failures.extend(validate(Path(arg)))
        if failures:
            for f in failures:
                print(f"FAIL: {f}")
            return 1
        print(f"OK: {len(argv) - 1} Fly config(s) validated")
        return 0

    if __name__ == "__main__":
        sys.exit(main(sys.argv))
    ```

    **`backend/scripts/r2_lifecycle.json`** — T-06-05 mitigation via **prefix-scoped GC of abandoned pre-commit uploads only**. Plan 05's commit handler uses a two-step R2 layout: presigned PUT targets `pending/{scan_id}.json`; after the sync commit transaction succeeds (HEAD + validate + DB insert + Stripe meter), the handler issues CopyObject → `teams/{team_id}/scans/{scan_id}.json` and DeleteObject on the `pending/` source key. The lifecycle rule therefore only ever expires objects that were PUT but never committed — committed scans at `teams/{team_id}/scans/` are NEVER touched by lifecycle. Blast radius bounded to abandoned uploads, fully delivers D-11.
    ```json
    {
      "rules": [
        {
          "id": "expire-abandoned-pending-uploads",
          "enabled": true,
          "conditions": {"prefix": "pending/"},
          "deleteObjectsTransition": {
            "condition": {"maxAge": {"unit": "days", "value": 7}}
          },
          "abortMultipartUploadsTransition": {
            "condition": {"maxAge": {"unit": "days", "value": 1}}
          }
        }
      ]
    }
    ```
    **Why this shape works:** the `pending/` prefix is write-only from Plan 05's presigned PUT; successful commits copy-and-delete out of it. Any object that lingers ≥7 days under `pending/` is by definition an abandoned upload (client disconnected, commit handler crashed pre-copy, malicious PUT never followed by commit). Deleting it reclaims R2 storage without risk to committed scans. T-06-05 is fully mitigated for Phase 6 (commit-HEAD size cap + lifecycle-scoped orphan GC both active).
