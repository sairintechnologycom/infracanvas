<objective>
Deliver the scan ingest pipeline end-to-end: two-step upload (POST /v1/scans → presigned PUT), atomic commit (POST /v1/scans/{id}/commit — size check + validate + insert + Stripe meter + rollback-on-failure), and retrieval (GET /v1/scans/{id} → presigned GET). Satisfies API-04, API-06, API-07, and TMM-02 (Stripe Billing Meters).

Purpose: This is the hot path the entire SaaS product turns on. Per D-09, billing consistency is load-bearing: if the meter event fails, the DB commit must roll back. The ordering (HEAD → validate → INSERT → Stripe → tx-commit) is chosen so Stripe is the LAST statement — meter failure rollbacks are safe, Stripe-success followed by DB-commit-failure has near-zero probability. Research callout #2: Content-Length-Range is NOT achievable on R2 presigned PUT — size enforcement happens at commit-time HEAD, NOT at URL sign time.

Output: Working upload → PUT → commit → retrieve flow. Tests assert: cross-team 404, 413 on oversized, 422 on bad ResourceGraph JSON, idempotent meter events (same scan_id twice → one meter event via identifier+idempotency_key dual layer), rollback on Stripe failure.
</objective>

    ```python
    from __future__ import annotations
    from functools import lru_cache
    from typing import Any
    import boto3
    from botocore.config import Config
    from app.settings import settings

    @lru_cache(maxsize=1)
    def get_r2_client():
        return boto3.client(
            "s3",
            endpoint_url=f"https://{settings.r2_account_id}.r2.cloudflarestorage.com",
            aws_access_key_id=settings.r2_access_key_id,
            aws_secret_access_key=settings.r2_secret_access_key,
            region_name="auto",
            config=Config(signature_version="s3v4"),  # REQUIRED for R2
        )

    def presigned_put(key: str, content_type: str = "application/json",
                       expires_in: int = 600) -> str:
        # NOTE (research callout #2): R2 does NOT support Content-Length-Range on
        # presigned PUT. Size enforcement happens at commit-time HEAD. Do NOT add
        # a Content-Length-Range Params entry here.
        return get_r2_client().generate_presigned_url(
            "put_object",
            Params={"Bucket": settings.r2_bucket, "Key": key, "ContentType": content_type},
            ExpiresIn=expires_in,
            HttpMethod="PUT",
        )

    def presigned_get(key: str, expires_in: int = 300) -> str:
        return get_r2_client().generate_presigned_url(
            "get_object",
            Params={"Bucket": settings.r2_bucket, "Key": key},
            ExpiresIn=expires_in,
        )

    def head(key: str) -> dict[str, Any]:
        return get_r2_client().head_object(Bucket=settings.r2_bucket, Key=key)

    def get_bytes(key: str) -> bytes:
        obj = get_r2_client().get_object(Bucket=settings.r2_bucket, Key=key)
        return obj["Body"].read()

    def copy(src_key: str, dst_key: str) -> None:
        # Server-side copy inside the same R2 bucket. Commit flow uses this to
        # move pending/{scan_id}.json → teams/{team_id}/scans/{scan_id}.json
        # after the sync DB+Stripe transaction succeeds. Delivers D-11 + T-06-05.
        get_r2_client().copy_object(
            Bucket=settings.r2_bucket,
            Key=dst_key,
            CopySource={"Bucket": settings.r2_bucket, "Key": src_key},
        )

    def delete(key: str) -> None:
        get_r2_client().delete_object(Bucket=settings.r2_bucket, Key=key)
    ```

    """Denormalized graph summary computation — shared between CLI and backend indexer."""
    from __future__ import annotations
    from infracanvas.graph.models import GraphSummary, ResourceGraph

    _SCORE_WEIGHTS = {"critical": 20, "high": 10, "medium": 5, "info": 1}

    def compute_summary(graph: ResourceGraph) -> GraphSummary:
        counts: dict[str, int] = {"critical": 0, "high": 0, "medium": 0, "info": 0}
        for node in graph.nodes:
            for f in node.findings:
                counts[f.severity.value] = counts.get(f.severity.value, 0) + 1
        score = 100 - sum(counts[k] * w for k, w in _SCORE_WEIGHTS.items())
        score = max(0, score)
        return GraphSummary(
            total_resources=len(graph.nodes),
            findings=counts,
            estimated_monthly_cost=0.0,   # preserved from CLI default; CostLens phases populate
            score=score,
        )
    ```

    _MAX_BYTES = 25 * 1024 * 1024
    _PUT_TTL_SECONDS = 600
    _GET_TTL_SECONDS = 300

    router = APIRouter(prefix="/v1/scans", tags=["scans"])
    _log = structlog.get_logger("app.scans")

    def _pending_key(scan_id: UUID) -> str:
        # Two-step layout: presigned PUT targets pending/; committed scans get copied
        # to teams/{team_id}/scans/ and the pending/ source deleted. This makes the
        # 7-day lifecycle rule (Plan 08, r2_lifecycle.json) safe: only abandoned
        # pre-commit uploads are ever GC'd. Delivers CONTEXT.md D-11 without risk
        # to committed scans.
        return f"pending/{scan_id}.json"

    def _final_key(team_id: UUID, scan_id: UUID) -> str:
        return f"teams/{team_id}/scans/{scan_id}.json"

    @router.post("", response_model=ScanCreateResp, status_code=200)
    async def create_scan(
        body: ScanCreateReq,
        principal: ClerkPrincipal = Depends(require_role("owner", "admin", "member")),
        team: Team = Depends(resolve_team_from_clerk_org),
    ) -> ScanCreateResp:
        """Step 1 of two-step upload. No R2 bytes cross the API. PUT target is
        pending/{scan_id}.json — commit step copies to teams/{team_id}/scans/…
        and deletes the pending/ source (see commit_scan)."""
        scan_id = new_uuid7()
        put_key = _pending_key(scan_id)
        url = await run_in_threadpool(
            r2.presigned_put, put_key, body.content_type, _PUT_TTL_SECONDS)
        expires_at = datetime.now(timezone.utc) + timedelta(seconds=_PUT_TTL_SECONDS)
        _log.info("scan_create", scan_id=str(scan_id), put_key=put_key)
        return ScanCreateResp(scan_id=scan_id, presigned_put_url=url, expires_at=expires_at)

    @router.post("/{scan_id}/commit", response_model=ScanGetResp, status_code=200)
    async def commit_scan(
        scan_id: UUID,
        body: ScanCommitReq,
        principal: ClerkPrincipal = Depends(require_role("owner", "admin", "member")),
        team: Team = Depends(resolve_team_from_clerk_org),
    ) -> ScanGetResp:
        """Atomic commit per D-09 / RESEARCH § F8 + D-11 two-step R2 copy. Order:
           1. HEAD pending/ → 404 if missing, 413 if > 25MB
           2. Fetch pending/ + ResourceGraph.model_validate_json → 422 on failure
           3. INSERT scans row (r2_key = final teams/ key)
           4. Stripe meter event (LAST in tx — failure rolls back DB)
           5. Commit (session.begin() context exit)
           6. Post-commit R2 moves: CopyObject pending/→teams/…, DeleteObject pending/
           7. Post-commit: fire-and-forget taskiq enqueue with request_id label
        """
        pending = _pending_key(scan_id)
        final = _final_key(team.id, scan_id)

        # 1. HEAD pending/
        try:
            head_meta = await run_in_threadpool(r2.head, pending)
        except ClientError as e:
            code = e.response.get("Error", {}).get("Code", "")
            if code in ("404", "NoSuchKey", "NotFound"):
                raise HTTPException(status.HTTP_404_NOT_FOUND, "object_not_found") from e
            raise
        size = int(head_meta["ContentLength"])
        if size > _MAX_BYTES:
            raise HTTPException(status.HTTP_413_REQUEST_ENTITY_TOO_LARGE,
                                 {"error": "too_large", "size_bytes": size, "max": _MAX_BYTES})

        # 2. Fetch pending/ + Pydantic validate against ResourceGraph
        blob = await run_in_threadpool(r2.get_bytes, pending)
        try:
            ResourceGraph.model_validate_json(blob)
        except ValidationError as e:
            raise HTTPException(status.HTTP_422_UNPROCESSABLE_ENTITY,
                                 {"errors": e.errors()[:10]}) from e

        # 3 + 4 + 5: tx begin → SET LOCAL → INSERT (with final key) → Stripe → COMMIT
        sm = get_sessionmaker()
        async with sm() as session:
            async with session.begin():
                await session.execute(
                    text("SET LOCAL app.current_team_id = :t"), {"t": str(team.id)})
                scan = Scan(
                    id=scan_id, team_id=team.id, r2_key=final,
                    sha256=body.sha256, size_bytes=size, status=ScanStatus.ready,
                )
                session.add(scan)
                await session.flush()  # surface UNIQUE violations before Stripe call

                # Stripe meter event — LAST inside tx (research callout #3)
                try:
                    await record_scan_meter_event(
                        scan_id=str(scan_id),
                        stripe_customer_id=team.stripe_customer_id or "",
                    )
                except stripe.error.StripeError as e:
                    _log.error("meter_event_failed", scan_id=str(scan_id), error=repr(e))
                    raise HTTPException(status.HTTP_502_BAD_GATEWAY, "meter_failed") from e
            # session.begin() exits → COMMIT here on success, ROLLBACK if exception above

        # 5b. Post-commit R2 move: copy pending/ → teams/{team_id}/scans/… then
        # delete pending/. DB is already committed with r2_key=final; if the
        # copy/delete fails we leave the pending/ object around (lifecycle rule
        # will GC it after 7d) and return 200 to the client — the scans row's
        # r2_key already points at `final`, so a subsequent GET would 500 until
        # reconciliation. Log at WARNING so on-call sees it; a follow-up
        # reconciler task (deferred Phase 7) can retry the copy.
        try:
            await run_in_threadpool(r2.copy, src_key=pending, dst_key=final)
            await run_in_threadpool(r2.delete, pending)
        except ClientError as e:
            _log.warning("r2_post_commit_copy_failed",
                         scan_id=str(scan_id), pending=pending, final=final,
                         error=repr(e))

        # 6. Post-commit — enqueue indexing (lazy import; Plan 06 provides)
        try:
            from app.queue.tasks.indexing import enqueue_scan_indexing
            rid = structlog.contextvars.get_contextvars().get("request_id", "")
            await enqueue_scan_indexing.kicker().with_labels(request_id=rid) \
                                                 .kiq(scan_id=str(scan_id))
        except Exception as e:  # enqueue failures must NOT reverse the commit
            _log.warning("indexing_enqueue_failed", scan_id=str(scan_id), error=repr(e))

        # Build response (fresh read inside team-scoped session)
        async with sm() as session:
            async with session.begin():
                await session.execute(
                    text("SET LOCAL app.current_team_id = :t"), {"t": str(team.id)})
                row = (await session.execute(
                    select(Scan).where(Scan.id == scan_id))).scalar_one()
                get_url = await run_in_threadpool(r2.presigned_get, row.r2_key, _GET_TTL_SECONDS)
                return ScanGetResp(
                    id=row.id, team_id=row.team_id, status=row.status,
                    presigned_get_url=get_url,
                    size_bytes=row.size_bytes, created_at=row.created_at,
                    summary_json=row.summary_json,
                )

    @router.get("/{scan_id}", response_model=ScanGetResp)
    async def get_scan(
        scan_id: UUID,
        principal: ClerkPrincipal = Depends(require_role("owner", "admin", "member", "basic_member")),
        team: Team = Depends(resolve_team_from_clerk_org),
        session: AsyncSession = Depends(team_scoped_session),
    ) -> ScanGetResp:
        row = (await session.execute(
            select(Scan).where(Scan.id == scan_id))).scalar_one_or_none()
        if row is None:
            # D-10: return 404 (not 403) on cross-team — don't leak existence
            raise HTTPException(status.HTTP_404_NOT_FOUND, "scan_not_found")
        get_url = await run_in_threadpool(r2.presigned_get, row.r2_key, _GET_TTL_SECONDS)
        return ScanGetResp(
            id=row.id, team_id=row.team_id, status=row.status,
            presigned_get_url=get_url,
            size_bytes=row.size_bytes, created_at=row.created_at,
            summary_json=row.summary_json,
        )
    ```
