    ```python
    from __future__ import annotations
    from typing import Any
    import sentry_sdk
    import structlog
    from taskiq import TaskiqMiddleware
    from taskiq.message import TaskiqMessage
    from taskiq.result import TaskiqResult

    _log = structlog.get_logger("app.worker")

    class RequestIdMiddleware(TaskiqMiddleware):
        """Rebind request_id (and scan_id if present) from task labels into
        structlog contextvar so worker-side logs share the HTTP trace.
        Implements D-21."""
        async def pre_execute(self, message: TaskiqMessage) -> TaskiqMessage:
            rid = message.labels.get("request_id", "") if message.labels else ""
            structlog.contextvars.clear_contextvars()
            if rid:
                structlog.contextvars.bind_contextvars(request_id=rid)
            # Convenience: many tasks carry scan_id as first positional arg
            if message.args and isinstance(message.args[0], str):
                structlog.contextvars.bind_contextvars(scan_id=message.args[0])
            return message

        async def post_execute(self, message: TaskiqMessage, result: TaskiqResult[Any]) -> None:
            structlog.contextvars.clear_contextvars()

    class DLQLogMiddleware(TaskiqMiddleware):
        """When a task exhausts retries, emit a `dlq=true` structured log line.
        Phase 6 MVP uses log-as-DLQ (RESEARCH § F7); a Redis-list DLQ can be Phase 8.
        """
        async def on_error(self, message: TaskiqMessage, result: TaskiqResult[Any],
                            exception: BaseException) -> None:
            labels = message.labels or {}
            retry_count = int(labels.get("_retry_count", 0) or 0)
            max_retries = int(labels.get("max_retries", 3) or 3)
            if retry_count >= max_retries:
                _log.error(
                    "task_dlq",
                    dlq=True,
                    task_name=message.task_name,
                    args=message.args,
                    kwargs=message.kwargs,
                    error=repr(exception),
                    request_id=labels.get("request_id", ""),
                )

    class SentryTaskMiddleware(TaskiqMiddleware):
        """Tag Sentry scope with task_name + request_id; capture exceptions on failure."""
        async def pre_execute(self, message: TaskiqMessage) -> TaskiqMessage:
            sentry_sdk.set_tag("task_name", message.task_name)
            labels = message.labels or {}
            if labels.get("request_id"):
                sentry_sdk.set_tag("request_id", labels["request_id"])
            return message

        async def on_error(self, message: TaskiqMessage, result: TaskiqResult[Any],
                            exception: BaseException) -> None:
            sentry_sdk.capture_exception(exception)
    ```

    ```python
    """enqueue_scan_indexing — post-commit denormalization writing scans.summary_json.

    Uses shared compute_summary() from cli/infracanvas/graph/summary.py so the
    CLI's scoring rubric and the backend's indexer remain a single source of truth.
    """
    from __future__ import annotations
    import asyncio
    import uuid
    from typing import Any
    import structlog
    from sqlalchemy import select, text, update
    from botocore.exceptions import ClientError
    from pydantic import ValidationError

    from infracanvas.graph.models import ResourceGraph
    from infracanvas.graph.summary import compute_summary

    from app.db.models import Scan
    from app.db.session import get_sessionmaker
    from app.queue.broker import broker
    from app.storage import r2

    _log = structlog.get_logger("app.worker.indexing")

    @broker.task(retry_on_error=True, max_retries=3, delay=5)
    async def enqueue_scan_indexing(scan_id: str) -> None:
        """Read scans row by id (raw_session since we don't yet know team),
        determine team_id, open team-scoped session, re-read + update summary.
        """
        scan_uuid = uuid.UUID(scan_id)
        sm = get_sessionmaker()

        # Phase 1: look up team_id. We need SOMETHING to do this without RLS
        # blocking; use a short tx that SETs LOCAL to the row's team_id after
        # a two-step discovery. Simpler: teams_lookup_by_org policy (Plan 04)
        # doesn't help here — we need scans.team_id, not teams.id. Fix: look up
        # scans via a transient admin-mode connection. OR: use the raw_session
        # to run a SECURITY DEFINER helper.
        # PHASE 6 CHOICE: use session.execute(text("SET LOCAL app.current_team_id = '<any uuid>'"))
        # with a special "indexer" value — but that defeats RLS.
        # CORRECT FIX: worker process connects via DATABASE_URL_MIGRATOR (owner role).
        # If DATABASE_URL_MIGRATOR is unset, fall back to a direct lookup by first
        # opening a session with `SET ROLE infracanvas_indexer` — deferred to Plan 08.
        # PRAGMATIC FIX FOR PHASE 6: worker role is the MIGRATOR (owner). `FORCE RLS`
        # still applies to owner. To bypass for the indexer path, the migrations
        # create a SECURITY DEFINER function `scan_team_id(uuid) RETURNS uuid`.
        #
        # For this plan, add a migration that creates the helper:
        # CREATE FUNCTION scan_team_id(p_scan_id uuid) RETURNS uuid
        #   LANGUAGE sql SECURITY DEFINER AS
        # $$ SELECT team_id FROM scans WHERE id = p_scan_id $$;
        # GRANT EXECUTE ON FUNCTION scan_team_id(uuid) TO infracanvas_app;
        # (Added as migration 004 in this task — see below.)

        async with sm() as session:
            async with session.begin():
                row = await session.execute(
                    text("SELECT scan_team_id(:s)"), {"s": str(scan_uuid)})
                team_id = row.scalar_one_or_none()
                if team_id is None:
                    _log.warning("scan_missing_for_indexing", scan_id=scan_id)
                    return

        # Phase 2: team-scoped work — download R2, validate, compute, UPDATE
        async with sm() as session:
            async with session.begin():
                await session.execute(
                    text("SET LOCAL app.current_team_id = :t"), {"t": str(team_id)})
                scan_row = (await session.execute(
                    select(Scan).where(Scan.id == scan_uuid))).scalar_one()
                blob = await asyncio.to_thread(r2.get_bytes, scan_row.r2_key)
                try:
                    graph = ResourceGraph.model_validate_json(blob)
                except ValidationError as e:
                    _log.error("indexing_validation_failed", scan_id=scan_id,
                                errors=e.errors()[:5])
                    return
                summary = compute_summary(graph)
                await session.execute(
                    update(Scan).where(Scan.id == scan_uuid).values(
                        summary_json=summary.model_dump()))
                _log.info("indexing_complete", scan_id=scan_id,
                           total_resources=summary.total_resources, score=summary.score)
    ```
