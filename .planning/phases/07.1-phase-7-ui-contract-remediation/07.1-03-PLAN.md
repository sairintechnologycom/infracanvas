2. Rewrite `dashboard/app/(dashboard)/settings/layout.tsx`:

```tsx
'use client'
import { Tabs, TabsList, TabsTrigger } from '@/components/ui/tabs'
import Link from 'next/link'
import { usePathname } from 'next/navigation'

const TAB_ROUTES = [
  { value: 'members', label: 'Members', href: '/settings/members' },
  { value: 'billing', label: 'Billing', href: '/settings/billing' },
  { value: 'integrations', label: 'Integrations', href: '/settings/integrations' },
] as const

export default function SettingsLayout({ children }: { children: React.ReactNode }) {
  const pathname = usePathname() ?? ''
  const active =
    TAB_ROUTES.find(t => pathname.startsWith(t.href))?.value ?? 'members'

  return (
    <div className="max-w-7xl mx-auto px-8 py-8">
      <h1 className="text-base font-semibold mb-6">Settings</h1>
      <Tabs value={active} className="w-full">
        <TabsList>
          {TAB_ROUTES.map(t => (
            <TabsTrigger key={t.value} value={t.value} asChild>
              <Link href={t.href}>{t.label}</Link>
            </TabsTrigger>
          ))}
        </TabsList>
      </Tabs>
      <div className="mt-6">{children}</div>
    </div>
  )
}
```

Notes:
- `asChild` on TabsTrigger lets us keep the Next.js `<Link>` for client-side navigation (no full page reload) while still getting `role="tab"` semantics.
- Active tab is computed from the pathname (no client state needed — Next.js handles it).
- Heading is `text-base` (UI-SPEC §Typography — never `text-xl`).
