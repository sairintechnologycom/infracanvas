Skeleton (adapt names to whatever the file already uses for `require_auth`, `get_db`, the ORM model class, and SQL helpers):

```python
@router.get(
    "/scans/{scan_id}/share-links",
    response_model=ShareLinkListResp,
    status_code=200,
)
async def list_share_links(
    scan_id: UUID,
    auth: AuthContext = Depends(require_auth),
    db: AsyncSession = Depends(get_db),
) -> ShareLinkListResp:
    # 1. Verify the scan belongs to caller's team (RLS gate — same pattern as DELETE).
    scan_q = select(Scan).where(Scan.id == scan_id, Scan.team_id == auth.team_id)
    scan = (await db.execute(scan_q)).scalar_one_or_none()
    if scan is None:
        raise HTTPException(status_code=404, detail="Scan not found")

    # 2. Fetch active (non-revoked, non-expired) share-links.
    now = datetime.now(timezone.utc)
    q = (
        select(ShareLink)
        .where(
            ShareLink.scan_id == scan_id,
            ShareLink.team_id == auth.team_id,
            ShareLink.revoked_at.is_(None),
            or_(ShareLink.expires_at.is_(None), ShareLink.expires_at > now),
        )
        .order_by(ShareLink.created_at.desc())
    )
    rows = (await db.execute(q)).scalars().all()

    return ShareLinkListResp(
        links=[
            ShareLinkListItem(
                id=str(row.id),
                expires_at=row.expires_at,
                created_by=row.created_by,
                has_password=row.password_hash is not None,
                created_at=row.created_at,
            )
            for row in rows
        ]
    )
```

import { NextRequest, NextResponse } from 'next/server'
import { backendFetch } from '@/lib/backend'

export async function GET(req: NextRequest) {
  const scanId = req.nextUrl.searchParams.get('scan_id')
  if (!scanId) {
    return NextResponse.json({ error: 'scan_id required' }, { status: 400 })

    cache: 'no-store',

}
