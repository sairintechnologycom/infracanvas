```ts
import { describe, it, expect } from 'vitest'
import { readdirSync, readFileSync, statSync } from 'node:fs'
import { join } from 'node:path'

const ROOT = join(__dirname, '..')
const ROOTS = ['app', 'components']

function* walk(dir: string): Generator<string> {
  for (const ent of readdirSync(dir)) {
    const p = join(dir, ent)
    if (statSync(p).isDirectory()) {
      // Skip components/ui/ — those are vendored shadcn primitives; the spec exempts them.
      if (p.endsWith('/components/ui') || p.endsWith('\\components\\ui')) continue
      yield* walk(p)
    } else if (/\.(tsx?|jsx?)$/.test(ent)) {
      yield p
    }
  }
}

function allSources(): { path: string; src: string }[] {
  const out: { path: string; src: string }[] = []
  for (const r of ROOTS) {
    for (const file of walk(join(ROOT, r))) {
      out.push({ path: file, src: readFileSync(file, 'utf8') })
    }
  }
  return out
}

const SOURCES = allSources()
const HOME = readFileSync(join(ROOT, 'app', '(dashboard)', 'page.tsx'), 'utf8')

function findHits(re: RegExp): string[] {
  const hits: string[] = []
  for (const { path, src } of SOURCES) {
    if (re.test(src)) hits.push(path)
  }
  return hits
}

describe('Polish drift gates (RMD-06)', () => {
  it('no text-xl headings outside vendored shadcn', () => {
    expect(findHits(/\btext-xl\b/)).toEqual([])
  })

  it('no text-lg headings outside vendored shadcn', () => {
    expect(findHits(/\btext-lg\b/)).toEqual([])
  })

  it('no bg-amber-500 or bg-amber-600 (CTA shade drift)', () => {
    expect(findHits(/\bbg-amber-(500|600)\b/)).toEqual([])
  })

  it('no ring-amber-* (focus-ring drift)', () => {
    expect(findHits(/\bring-amber-/)).toEqual([])
  })

  it('no bg-amber-50 decorative chips', () => {
    expect(findHits(/\bbg-amber-50\b/)).toEqual([])
  })

  it('no text-amber-600 on text-links (amber overuse)', () => {
    // Reserved-for list does NOT include text-link text-amber-600.
    // The grade-C pill uses text-amber-700; we allow that.
    expect(findHits(/\btext-amber-600\b/)).toEqual([])
  })

  it('home page uses px-8 py-12 gap-12 gutters', () => {
    expect(HOME).toMatch(/px-8/)
    expect(HOME).toMatch(/py-12/)
    expect(HOME).toMatch(/gap-12/)
  })

  it('home page does not render an unspec "Overview" h1', () => {
    expect(HOME).not.toMatch(/<h1[^>]*>\s*Overview\s*</)
  })
})
```

```bash
#!/usr/bin/env bash
set -euo pipefail
cd "$(dirname "$0")/.."

fail=0

check() {
  local desc="$1" expected="$2"; shift 2
  local got
  got=$("$@" || true)
  if [[ "$got" != "$expected" ]]; then
    echo "FAIL: $desc — expected '$expected', got '$got'"
    fail=1
  else
    echo "ok   $desc"
  fi
}

# Exclude components/ui/ (vendored shadcn primitives)
EXCL='--exclude-dir=ui'

check "no text-xl"   "0" bash -c "grep -rn $EXCL 'text-xl' app components | wc -l | tr -d ' '"
check "no text-lg"   "0" bash -c "grep -rn $EXCL 'text-lg' app components | wc -l | tr -d ' '"
check "no bg-amber-500/600" "0" bash -c "grep -rEn $EXCL 'bg-amber-(500|600)' app components | wc -l | tr -d ' '"
check "no ring-amber" "0" bash -c "grep -rn $EXCL 'ring-amber' app components | wc -l | tr -d ' '"
check "no bg-amber-50" "0" bash -c "grep -rn $EXCL 'bg-amber-50' app components | wc -l | tr -d ' '"
check "no text-amber-600" "0" bash -c "grep -rn $EXCL 'text-amber-600' app components | wc -l | tr -d ' '"

# Home gutters present
check "home px-8" "1" bash -c "grep -n 'px-8' 'app/(dashboard)/page.tsx' | wc -l | tr -d ' '"
check "home py-12" "1" bash -c "grep -n 'py-12' 'app/(dashboard)/page.tsx' | wc -l | tr -d ' '"
check "home gap-12" "1" bash -c "grep -n 'gap-12' 'app/(dashboard)/page.tsx' | wc -l | tr -d ' '"

# Overview h1 removed
check "no Overview h1" "0" bash -c "grep -nE '<h1[^>]*>\\s*Overview\\s*<' 'app/(dashboard)/page.tsx' | wc -l | tr -d ' '"

exit $fail
```
