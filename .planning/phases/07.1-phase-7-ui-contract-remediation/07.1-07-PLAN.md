2. Create `dashboard/components/layout/TopBarActions.tsx` per RESEARCH §Pattern 4 verbatim:

```tsx
'use client'
import { createContext, useContext, useState, useCallback, type ReactNode } from 'react'

type ActionsContextShape = {
  actions: ReactNode
  set: (jsx: ReactNode) => void
  clear: () => void
}

const ActionsContext = createContext<ActionsContextShape | null>(null)

export function TopBarActionsProvider({ children }: { children: ReactNode }) {
  const [actions, setActions] = useState<ReactNode>(null)
  const set = useCallback((jsx: ReactNode) => setActions(jsx), [])
  const clear = useCallback(() => setActions(null), [])
  return <ActionsContext.Provider value={{ actions, set, clear }}>{children}</ActionsContext.Provider>
}

export function useTopBarActions() {
  const ctx = useContext(ActionsContext)
  if (!ctx) throw new Error('useTopBarActions must be inside TopBarActionsProvider')
  return ctx
}

export function TopBarActionsSlot() {
  const ctx = useContext(ActionsContext)
  return <>{ctx?.actions ?? null}</>
}
```

The breadcrumb has two responsibilities:
- Static segment lookup table: known route prefixes get fixed labels (`/scans` -> "Scans", `/settings` -> "Settings", etc.).
- Dynamic segments (UUIDs): rendered via a label-setter context that scan-detail pages populate (similar pattern to TopBarActions). For Phase 7.1 scope, a simpler approach is acceptable: detect when the second segment looks like a UUID (regex `/^[0-9a-f]{8}-[0-9a-f]{4}-/`) and render a placeholder like "Scan" or render nothing for that segment — the page itself can set a label via a parallel TopBarLabel context if you want full UI-SPEC fidelity.

const LABELS: Record<string, string> = {
  '': 'Home',
  scans: 'Scans',
  settings: 'Settings',
  billing: 'Billing',
  members: 'Members',
  integrations: 'Integrations',
  compare: 'Compare',
}

const UUID_RE = /^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$/i

function labelFor(seg: string): string | null {
  if (LABELS[seg] != null) return LABELS[seg]
  if (UUID_RE.test(seg)) return null  // dynamic UUID — handled by detail-page label context, or rendered empty for now
  // fallback for unknown segments: sentence-case (capitalize first letter only)
  return seg.charAt(0).toUpperCase() + seg.slice(1)
}

// in JSX:
const segments = (pathname ?? '').split('/').filter(Boolean)
const crumbs = segments.map(labelFor).filter((s): s is string => s !== null)
