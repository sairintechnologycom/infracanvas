2. Create `cli/infracanvas/parser/module.py`:
   ```python
   """Recursive Terraform module resolution."""
   from __future__ import annotations
   from pathlib import Path
   from infracanvas.parser.hcl import parse_directory, ParsedTerraform

   def resolve_modules(directory: Path, parsed: ParsedTerraform, depth: int = 0) -> None:
       if depth >= 3:
           return
       for module_block in parsed._raw_modules:
           source = module_block.get("source", "")
           name = module_block.get("__name__", "unknown")
           if not (source.startswith("./") or source.startswith("../")):
               continue  # skip registry/git sources
           module_dir = (directory / source).resolve()
           if not module_dir.is_dir():
               continue
           sub_parsed = parse_directory(module_dir)
           for res in sub_parsed.resources:
               res.module = f"module.{name}"
           parsed.resources.extend(sub_parsed.resources)
           resolve_modules(module_dir, sub_parsed, depth + 1)
   ```
