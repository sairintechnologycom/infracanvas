import { describe, it, expect } from 'vitest';
import { render, screen } from '@testing-library/react';
import { ReactFlowProvider } from '@xyflow/react';

// Minimal ResourceNodeData fixture — shape follows viewer/src/types.ts.
// If the real type has additional required fields, extend this stub accordingly.
function makeNodeData(overrides: Record<string, unknown> = {}) {
  return {
    id: 'aws_vpc.main',
    type: 'aws_vpc',
    name: 'main',
    provider: 'aws',
    module: '',
    region: 'us-east-1',
    group: 'network',
    attributes: {},
    findings: [],
    dependencies: [],
    drift: 'unchanged',
    ...overrides,
  };
}

// ResourceNode is a ReactFlow custom node — wrap in ReactFlowProvider to satisfy Handle usage.
function renderNode(data: ReturnType<typeof makeNodeData>) {
  const nodeProps = {
    id: data.id,
    data,
    selected: false,
    type: 'resource',
    zIndex: 0,
    isConnectable: false,
    xPos: 0,
    yPos: 0,
    dragging: false,
  } as unknown as Parameters<typeof ResourceNodeMemo>[0];

  return render(
    <ReactFlowProvider>
      <ResourceNodeMemo {...nodeProps} />
    </ReactFlowProvider>
  );
}

describe('ResourceNode — Phase 5.1 badges', () => {
  it('5.1-K: renders ⚠ unresolved marker when type is _infracanvas_unresolved_module', () => {
    renderNode(makeNodeData({
      type: '_infracanvas_unresolved_module',
      name: 'broken',

    const marker = screen.getByTestId('resource-node-unresolved-marker');

  it('5.1-M: renders ×? badge when attributes._unresolved_count is true', () => {
    renderNode(makeNodeData({
