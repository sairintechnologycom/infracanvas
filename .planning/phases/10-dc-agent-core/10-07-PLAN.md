Retry contract (D-07):
  - 3 total attempts (initial + 2 retries)
  - Backoff: 2s before retry 1, 4s before retry 2 (linear, not exponential)
  - Retry on transport error OR HTTP 5xx
  - Do NOT retry on 4xx (auth/validation failure — 401, 422)
  - On retry exhaustion: log WARN with site_id + endpoint + last error; return nil (drop-and-continue)

HTTP timeout: 15s per request — bounds a single attempt; combined with 2 retries + 6s backoff
gives a worst-case ~57s ceiling per push, well within the 5min routes ticker cadence.

    // Package push owns the agent's HTTP push client.
    //
    // D-07 contract: 3 attempts (1 + 2 retries), linear backoff 2s/4s, drop-and-continue
    // after exhaustion (logged at WARN; the daemon ticker fires again next interval).
    package push

    import (
        "bytes"
        "context"
        "encoding/json"
        "fmt"
        "io"
        "net/http"
        "strings"
        "time"

        "go.uber.org/zap"
    )

    const (
        defaultRequestTimeout = 15 * time.Second
        routesPath            = "/v1/agent/routes"
        flowsPath             = "/v1/agent/flows"
    )

    // BackoffFunc returns the wait duration BEFORE attempt N (1-indexed).
    // Production wiring uses linearBackoff (2s, 4s); tests inject a fast variant.
    type BackoffFunc func(attempt int) time.Duration

    func linearBackoff(attempt int) time.Duration {
        return time.Duration(attempt*2) * time.Second // attempt 1 -> 2s, attempt 2 -> 4s
    }

    type Client struct {
        backendURL string
        token      string
        http       *http.Client
        log        *zap.Logger
        backoff    BackoffFunc
    }

    func NewClient(backendURL, token string, log *zap.Logger) *Client {
        if log == nil {
            log = zap.NewNop()
        }
        return &Client{
            backendURL: strings.TrimRight(backendURL, "/"),
            token:      token,
            http:       &http.Client{Timeout: defaultRequestTimeout},
            log:        log,
            backoff:    linearBackoff,
        }
    }

    // SetBackoff replaces the production backoff with a custom one (test seam).
    func (c *Client) SetBackoff(b BackoffFunc) { c.backoff = b }

    // PushRoutes POSTs a routes batch with retry-twice-then-drop semantics (D-07).
    // Returns nil after 3 retried-failures (the batch is silently dropped, not surfaced
    // to the daemon ticker — drops are visible only via WARN logs and Axiom metrics).
    // Returns a non-nil error ONLY for non-retryable failures (4xx) so the caller
    // can surface auth/validation problems explicitly.
    func (c *Client) PushRoutes(ctx context.Context, p RoutesPayload) error {
        body, err := json.Marshal(p)
        if err != nil {
            return fmt.Errorf("push: marshal routes: %w", err)
        }
        return c.postWithRetry(ctx, routesPath, body, "routes",
            zap.String("site_id", p.SiteID),
            zap.String("device_host", p.DeviceHost),
            zap.Int("count", len(p.Routes)))
    }

    func (c *Client) PushFlows(ctx context.Context, p FlowsPayload) error {
        body, err := json.Marshal(p)
        if err != nil {
            return fmt.Errorf("push: marshal flows: %w", err)
        }
        return c.postWithRetry(ctx, flowsPath, body, "flows",
            zap.String("site_id", p.SiteID),
            zap.Int("count", len(p.Flows)))
    }

    func (c *Client) postWithRetry(
        ctx context.Context,
        path string,
        body []byte,
        kind string,
        logFields ...zap.Field,
    ) error {
        url := c.backendURL + path
        var lastErr error
        for attempt := 0; attempt < 3; attempt++ {
            if attempt > 0 {
                wait := c.backoff(attempt)
                select {
                case <-time.After(wait):
                case <-ctx.Done():
                    return ctx.Err()
                }
            }
            status, err := c.doPost(ctx, url, body)
            if err == nil {
                return nil // 2xx
            }
            // Auth/validation failures are non-retryable.
            if status == http.StatusUnauthorized || status == http.StatusUnprocessableEntity ||
                status == http.StatusForbidden {
                return err
            }
            lastErr = err
        }
        c.log.Warn("push_drop_after_retries",
            append(logFields,
                zap.String("kind", kind),
                zap.String("path", path),
                zap.Error(lastErr),
            )...)
        return nil
    }

    // doPost issues one POST attempt. Returns (statusCode, error). On 2xx returns
    // (status, nil). On non-2xx returns (status, error including statusCode + body).
    func (c *Client) doPost(ctx context.Context, url string, body []byte) (int, error) {
        req, err := http.NewRequestWithContext(ctx, http.MethodPost, url, bytes.NewReader(body))
        if err != nil {
            return 0, fmt.Errorf("push: build request: %w", err)
        }
        req.Header.Set("Authorization", "Bearer "+c.token)
        req.Header.Set("Content-Type", "application/json")
        resp, err := c.http.Do(req)
        if err != nil {
            return 0, fmt.Errorf("push: do: %w", err)
        }
        defer func() { _ = resp.Body.Close() }()
        if resp.StatusCode >= 200 && resp.StatusCode < 300 {
            return resp.StatusCode, nil
        }
        var sample bytes.Buffer
        _, _ = io.CopyN(&sample, resp.Body, 512) // cap snippet at 512 bytes
        return resp.StatusCode, fmt.Errorf("push: status=%d body=%s", resp.StatusCode, sample.String())
    }

    //
    // RouteCollectorFn dials a device and returns its current routes.
    // Production wiring binds the netconf/ssh/config-import implementations
    // by protocol; tests inject deterministic stubs.
    type RouteCollectorFn func(ctx context.Context, dev config.Device) ([]netconf.RouteRecord, error)

    // Pusher abstracts push.Client for tests.
    type Pusher interface {
        PushRoutes(ctx context.Context, p push.RoutesPayload) error
        PushFlows(ctx context.Context, p push.FlowsPayload) error
    }

    // collectorFor returns the RouteCollectorFn appropriate to the device protocol.
    // Returns nil for protocols not supported in this code path (e.g. BGP-only devices
    // pending Phase 11).
    func collectorFor(dev config.Device) RouteCollectorFn {
        switch dev.Protocol {
        case config.ProtocolNetconf:
            c := netconf.NewCollector(netconf.DefaultDialer())
            return c.GetRoutes
        case config.ProtocolSSH:
            c := ssh.NewCollector(ssh.DefaultDialer())
            return c.GetRoutes
        case config.ProtocolConfigImport:
            return func(ctx context.Context, d config.Device) ([]netconf.RouteRecord, error) {
                return config.LoadConfigImport(d.ConfigFile)
            }
        }
        return nil
    }

    // collectAndPushRoutes iterates devices, dials each, pushes results.
    // Pure function over (cfg, pusher, log) so tests can swap pusher and
    // override collectorFor by closing over a custom dispatch.
    func collectAndPushRoutes(ctx context.Context, cfg *config.Config, pusher Pusher, log *zap.Logger) {
        for _, dev := range cfg.Devices {
            fn := collectorFor(dev)
            if fn == nil {
                log.Debug("collector_skip_unsupported_protocol", zap.String("protocol", dev.Protocol))
                continue
            }
            routes, err := fn(ctx, dev)
            if err != nil {
                log.Warn("collect_routes_failed",
                    zap.String("device", dev.Host),
                    zap.String("protocol", dev.Protocol),
                    zap.Error(err))
                continue
            }
            payload := push.RoutesPayload{
                SiteID:      dev.SiteID,
                CollectedAt: time.Now().UTC().Format(time.RFC3339),
                DeviceHost:  dev.Host,
                Routes:      routes,
            }

                log.Warn("push_routes_failed",
                    zap.String("device", dev.Host),

    func flushFlowBuffer(ctx context.Context, cfg *config.Config, buf *netflow.RingBuffer, pusher Pusher, log *zap.Logger) {
