    ```python
    """Pure-function diff algorithm for ResourceGraph comparison.

    compute_diff(graph_a, graph_b) -> ResourceDiffResp

    Performs an outer-join on node IDs. "Changed" = any attribute key differs
    in value between the two graphs (D-12: same definition as the drift overlay).
    Edge diff is set-based on (source, target, relationship) tuples.

    Designed for reuse by:
    - GET /v1/scans/{a}/compare/{b} (Phase 7)
    - CLI infracanvas diff command (future)
    - PR-bot status checks (v1.2 PRB-01)
    """
    from __future__ import annotations

    from uuid import UUID

    from infracanvas.graph.models import ResourceGraph

    from app.schemas.scan import NodeDiff, ResourceDiffResp

    _MAX_NODES = 5000


    def compute_diff(
        graph_a: ResourceGraph,
        graph_b: ResourceGraph,
        scan_a_id: UUID,
        scan_b_id: UUID,
    ) -> ResourceDiffResp:
        """Diff two ResourceGraph instances and return a structured ResourceDiffResp.

        Node diff: outer-join on node.id.
        Edge diff: set diff on (source, target, relationship) tuples.
        """
        nodes_a: dict[str, dict] = {n.id: n.attributes for n in graph_a.nodes}
        nodes_b: dict[str, dict] = {n.id: n.attributes for n in graph_b.nodes}

        all_ids = set(nodes_a) | set(nodes_b)
        node_diffs: list[NodeDiff] = []

        counts = {"added": 0, "removed": 0, "changed": 0, "unchanged": 0}

        for node_id in sorted(all_ids):  # sorted for deterministic output
            in_a = node_id in nodes_a
            in_b = node_id in nodes_b

            if in_b and not in_a:
                kind = "added"
                counts["added"] += 1
                node_diffs.append(
                    NodeDiff(
                        id=node_id,
                        kind="added",
                        before=None,
                        after=nodes_b[node_id],
                    )
                )
            elif in_a and not in_b:
                kind = "removed"
                counts["removed"] += 1
                node_diffs.append(
                    NodeDiff(
                        id=node_id,
                        kind="removed",
                        before=nodes_a[node_id],
                        after=None,
                    )
                )
            else:
                # Both present — compare attributes
                attrs_a = nodes_a[node_id]
                attrs_b = nodes_b[node_id]
                changed_fields = _diff_attrs(attrs_a, attrs_b)
                if changed_fields:
                    counts["changed"] += 1
                    node_diffs.append(
                        NodeDiff(
                            id=node_id,
                            kind="changed",
                            before=attrs_a,
                            after=attrs_b,
                            changed_fields=changed_fields,
                        )
                    )
                else:
                    counts["unchanged"] += 1
                    node_diffs.append(
                        NodeDiff(
                            id=node_id,
                            kind="unchanged",
                            before=attrs_a,
                            after=attrs_b,
                        )
                    )

        # Cap to prevent oversized responses (D-11: diff ~tens of KB)
        node_diffs = node_diffs[:_MAX_NODES]

        # Edge diff: set-based on (source, target, relationship) tuples
        def edge_key(e) -> tuple[str, str, str]:  # type: ignore[no-untyped-def]
            return (e.source, e.target, e.relationship)

        edges_a = {edge_key(e): e for e in graph_a.edges}
        edges_b = {edge_key(e): e for e in graph_b.edges}

        edges_added = [
            {"source": e.source, "target": e.target, "relationship": e.relationship}
            for e in edges_b.values()
            if edge_key(e) not in edges_a
        ]
        edges_removed = [
            {"source": e.source, "target": e.target, "relationship": e.relationship}
            for e in edges_a.values()
            if edge_key(e) not in edges_b
        ]

        return ResourceDiffResp(
            scan_a_id=scan_a_id,
            scan_b_id=scan_b_id,
            nodes=node_diffs,
            edges_added=edges_added,
            edges_removed=edges_removed,
            summary=counts,
        )


    def _diff_attrs(attrs_a: dict, attrs_b: dict) -> list[str]:
        """Return list of attribute keys that differ between two attribute dicts.

        A key is "changed" if: present in one but not the other, or values differ.
        Uses == comparison (D-12: any attribute differs = changed).
        """
        all_keys = set(attrs_a) | set(attrs_b)
        return [k for k in sorted(all_keys) if attrs_a.get(k) != attrs_b.get(k)]
    ```

    ```python
    @router.get("/{scan_a_id}/compare/{scan_b_id}", response_model=ResourceDiffResp)
    async def compare_scans(
        scan_a_id: UUID,
        scan_b_id: UUID,
        principal: ClerkPrincipal = Depends(  # noqa: B008
            require_role("owner", "admin", "member", "basic_member")
        ),
        team: Team = Depends(resolve_team_from_clerk_org),
    ) -> ResourceDiffResp:
        """Diff two scans from the same team.

        Both scans must belong to the caller's team — RLS makes cross-team rows
        invisible, so missing rows return 404 (D-18: don't leak cross-team existence).
        R2 blobs are fetched concurrently via asyncio.gather to halve latency.
        """
        sm = get_sessionmaker()
        async with sm() as session:
            async with session.begin():
                await session.execute(
                    text("SELECT set_config('app.current_team_id', :t, true)"),
                    {"t": str(team.id)},
                )
                row_a = (
                    await session.execute(select(Scan).where(Scan.id == scan_a_id))
                ).scalar_one_or_none()
                row_b = (
                    await session.execute(select(Scan).where(Scan.id == scan_b_id))
                ).scalar_one_or_none()
                if row_a is None or row_b is None:
                    raise HTTPException(status.HTTP_404_NOT_FOUND, "scan_not_found")

        # Fetch both blobs concurrently — never sequential for two large blobs
        try:
            blob_a, blob_b = await asyncio.gather(
                run_in_threadpool(r2.get_bytes, row_a.r2_key),
                run_in_threadpool(r2.get_bytes, row_b.r2_key),
            )
        except ClientError as exc:
            code = exc.response.get("Error", {}).get("Code", "")
            if code in ("404", "NoSuchKey", "NotFound"):
                raise HTTPException(status.HTTP_404_NOT_FOUND, "object_not_found") from exc
            raise

        from infracanvas.graph.models import ResourceGraph  # local import — CLI package

        graph_a = ResourceGraph.model_validate_json(blob_a)
        graph_b = ResourceGraph.model_validate_json(blob_b)

        return compute_diff(graph_a, graph_b, scan_a_id=scan_a_id, scan_b_id=scan_b_id)
    ```
