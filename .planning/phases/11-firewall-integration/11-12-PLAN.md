```go
// collectAndPushFirewall is the 4th-ticker entry point. Mints one snapshot_id
// per device per tick (RESEARCH Pattern 2 — UUIDv4 shared across the 3 push
// endpoints; backend ON CONFLICT DO NOTHING on snapshot_id makes them
// independently idempotent regardless of arrival order).
func collectAndPushFirewall(ctx context.Context, cfg *config.Config, pusher Pusher, log *zap.Logger) {
    for _, dev := range cfg.Devices {
        pullFn := firewallCollectorFor(dev)
        if pullFn == nil {
            continue  // non-firewall protocol; route/bgp/flow tickers handle these
        }

        // Per-device 50min budget — leaves 10min buffer before next 1h tick
        devCtx, cancel := context.WithTimeout(ctx, 50*time.Minute)
        rules, nats, objs, err := pullFn(devCtx, dev)
        cancel()
        if err != nil {
            log.Warn("firewall_pull_failed",
                zap.String("host", dev.Host),
                zap.String("protocol", dev.Protocol),
                zap.Error(err))
            continue
        }

        snapshotID := uuid.NewString()
        snapshotTS := time.Now().UTC().Format(time.RFC3339)
        siteID := dev.SiteID  // empty = use agent's site_token default
        firewallID := dev.Host  // device identifier; future: serial number from device
        if dev.Protocol == config.ProtocolCheckpointImport {
            firewallID = filepath.Base(dev.ConfigFile)  // air-gapped — no host
        }
        vendor := firewallVendor(dev.Protocol)

        rulesPayload := push.FirewallRulesPayload{
            SiteID: siteID, SnapshotID: snapshotID, FirewallID: firewallID,
            Vendor: vendor, Source: dev.Protocol, SnapshotTS: snapshotTS,
            Rules: rules,
        }
        natPayload := push.FirewallNATPayload{
            SiteID: siteID, SnapshotID: snapshotID, FirewallID: firewallID,
            Vendor: vendor, Source: dev.Protocol, SnapshotTS: snapshotTS,
            NATRules: nats,
        }
        objsPayload := push.FirewallObjectsPayload{
            SiteID: siteID, SnapshotID: snapshotID, FirewallID: firewallID,
            Vendor: vendor, Source: dev.Protocol, SnapshotTS: snapshotTS,
            Objects: objs,
        }

        // Three sequential pushes. Push client owns retry-twice-then-drop (D-07).
        // Errors logged but do not abort other devices.
        if err := pusher.PushFirewallRules(ctx, rulesPayload); err != nil {
            log.Warn("firewall_rules_push_failed",
                zap.String("host", dev.Host), zap.String("snapshot_id", snapshotID),
                zap.Error(err))
        }
        if err := pusher.PushFirewallNAT(ctx, natPayload); err != nil {
            log.Warn("firewall_nat_push_failed",
                zap.String("host", dev.Host), zap.String("snapshot_id", snapshotID),
                zap.Error(err))
        }
        if err := pusher.PushFirewallObjects(ctx, objsPayload); err != nil {
            log.Warn("firewall_objects_push_failed",
                zap.String("host", dev.Host), zap.String("snapshot_id", snapshotID),
                zap.Error(err))
        }

        log.Info("firewall_pull_complete",
            zap.String("host", dev.Host),
            zap.String("protocol", dev.Protocol),
            zap.String("snapshot_id", snapshotID),
            zap.Int("rules", len(rules)),
            zap.Int("nats", len(nats)),
            zap.Int("objects", len(objs)))
    }
}
```
