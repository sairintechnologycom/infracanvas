import os
import structlog
from sqlalchemy import text

from app.broker import broker  # adjust import to actual project broker module
from app.db.session import get_sessionmaker

_log = structlog.get_logger("app.tasks.firewall_prune")


@broker.task  # adjust decorator to project convention; if periodic, use schedule
async def prune_firewall_snapshots() -> dict[str, int]:
    """Delete firewall_ruleset_snapshots older than TTL. Returns {'deleted': N}."""
    ttl_days = int(os.environ.get("FIREWALL_SNAPSHOT_TTL_DAYS", "14"))
    sm = get_sessionmaker()
    async with sm() as session, session.begin():
        # NOTE: prune runs as the infracanvas_app role with RLS — set app.current_team_id
        # to a sentinel value would NOT bypass RLS; instead the task runs in a privileged
        # context. If project has a taskiq superuser pattern, apply it here.
        # For now, iterate teams: SELECT team_id FROM teams; FOR EACH set GUC and DELETE.
        teams = (await session.execute(text("SELECT id FROM teams"))).scalars().all()
        total_deleted = 0
        for team_id in teams:
            await session.execute(
                text("SELECT set_config('app.current_team_id', :t, true)"),
                {"t": str(team_id)},
            )
            result = await session.execute(
                text(f"DELETE FROM firewall_ruleset_snapshots "
                     f"WHERE snapshot_ts < NOW() - INTERVAL '{ttl_days} days'"),
            )
            total_deleted += result.rowcount or 0
    _log.info("firewall_snapshots_pruned", deleted=total_deleted, ttl_days=ttl_days)
    return {"deleted": total_deleted}
