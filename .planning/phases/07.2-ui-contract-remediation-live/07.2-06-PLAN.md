    Implementation:

    ```ts
    import { NextResponse } from 'next/server'
    import { isUUID } from '@/lib/utils'
    import { backendFetch } from '@/lib/backend'
    import { fetchScanJson } from '@/lib/r2'
    import type { ScanGetResp, Finding } from '@/lib/types'

    export async function GET(req: Request) {
      const { searchParams } = new URL(req.url)
      const scanId = searchParams.get('scan_id')

      if (!scanId || !isUUID(scanId)) {
        return NextResponse.json({ error: 'invalid scan_id' }, { status: 400 })
      }

      // Step 1: get presigned URL via the same backend path /scans/[id] uses.
      let scanMeta: ScanGetResp
      try {
        scanMeta = await backendFetch<ScanGetResp>(`/v1/scans/${scanId}`)
      } catch (err) {
        // backendFetch throws Error('{status}') on non-2xx
        const msg = (err as Error).message
        if (msg === '404') {
          return NextResponse.json({ error: 'scan not found' }, { status: 404 })
        }
        throw err
      }

      if (!scanMeta.presigned_get_url) {
        return NextResponse.json({ error: 'scan not ready' }, { status: 404 })
      }

      // Step 2: fetch full ResourceGraph from R2.
      const graph = await fetchScanJson({
        presignedUrl: scanMeta.presigned_get_url,
        onPresignedExpired: async () => {
          const refreshed = await backendFetch<ScanGetResp>(`/v1/scans/${scanId}`)
          return refreshed.presigned_get_url ?? ''
        },
      })

      // graph.nodes[].findings is Finding[]. Aggregate findings across all nodes
      // (the ResourceGraph attaches findings per-node, not at top-level).
      const allFindings: Array<{ rule_id: string; title: string; resource_id: string }> = []
      for (const node of graph.nodes ?? []) {
        for (const f of node.findings ?? []) {
          if (f.severity === 'critical') {
            allFindings.push({
              rule_id: f.rule_id,
              title: f.title,
              resource_id: node.id,
            })
            if (allFindings.length >= 3) break
          }
        }
        if (allFindings.length >= 3) break
      }

      return NextResponse.json({ findings: allFindings })
    }
    ```

    Notes:
    - Severity is the literal lowercase string `'critical'` — confirmed by reading
      `cli/infracanvas/graph/models.py:11` (`critical = "critical"`). Do NOT use uppercase.
    - The scan ResourceGraph attaches findings per node (`node.findings: Finding[]`),
      not a flat top-level `findings` array — confirmed by Finding/ResourceNode type
      definitions in the viewer package re-exported via `dashboard/lib/types.ts`.
    - `resource_id` for the response comes from `node.id` (the dependency-graph node id).
    - If `isUUID` is not exported from `lib/utils.ts`, inline a regex fallback (per
      CONTEXT.md `isUUID()` was preserved in plan 07.1-01 specifically for T-07-08-01
      mitigation — STATE.md line 66). Read `dashboard/lib/utils.ts` first to confirm.
