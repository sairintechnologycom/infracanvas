async def recompute_paths_all_sites() -> dict[str, int]:
    """D-04 fan-out — enqueue per-site compute under each team's RLS context."""
    sm = get_sessionmaker()
    enqueued = 0
    async with sm() as session:
        team_rows = (await session.execute(text("SELECT id FROM teams"))).all()
        team_ids = [str(row[0]) for row in team_rows]
    for team_id in team_ids:
        async with sm() as session, session.begin():
            await session.execute(
                text("SELECT set_config('app.current_team_id', :t, true)"),
                {"t": team_id},
            )
            site_rows = (await session.execute(text("SELECT id FROM dc_sites"))).all()
        for (site_id,) in site_rows:
            await recompute_paths_for_site.kiq(site_id=site_id)
            enqueued += 1
    _log.info("path_recompute_fanout", sites_enqueued=enqueued, teams_scanned=len(team_ids))
    return {"enqueued": enqueued}

def _jsonify_hops(hops: list) -> str:
    return json.dumps([h.model_dump() if hasattr(h, "model_dump") else dict(h) for h in hops])


def _json_dumps(obj) -> str:
    return json.dumps(obj, default=str)


def _leg_routes(path, route_snapshot: dict[str, list]) -> list:
    """Warning 5 — return the route table of the LAST hop of `path`.

    The last hop is where the egress decision was made for that leg. If the
    hop's device is not in route_snapshot (e.g., compute fell off the
    snapshot edge), return an empty list — the classifier scorers handle
    empty inputs (score == 0.0). When both legs end on the same device
    (intra-DC asymmetry), fwd_routes and ret_routes are legitimately equal.
    """
    if not path.hops:
        return []
    last_node = path.hops[-1].node_id
    return route_snapshot.get(last_node, [])

def test_classify_route_leak_fires_when_legs_have_different_route_tables():
    """Warning 5 — cross-device asymmetry: forward leg and return leg's last
    hops are different devices, so fwd_routes and ret_routes have disjoint
    prefix sets. score_leak should exceed 0.4 and ROUTE_LEAK should be cause.
    """
    from app.security.pathcompute.classify import classify
    from infracanvas.graph.models import NetworkPath, PathHop
    fwd = NetworkPath(id="fwd", direction="forward",
                      hops=[PathHop(node_id="dev-a", source_ip="10.0.0.1", dest_ip="10.0.1.1")])
    ret = NetworkPath(id="ret", direction="return",
                      hops=[PathHop(node_id="dev-b", source_ip="10.0.1.1", dest_ip="10.0.0.1")])
    fwd_routes = [type("R", (), {"prefix": "10.0.0.0/24", "as_path": "65001 65002", "metric": 100})()]
    ret_routes = [type("R", (), {"prefix": "10.0.1.0/24", "as_path": "65001 65003 65999", "metric": 100})()]
    cause, conf, ev = classify(fwd, ret, fwd_routes, ret_routes, nat_rules=[])
    assert cause in ("ROUTE_LEAK", "BGP_LOCAL_PREF")  # both possible; both prove D-08 live
    assert conf >= 0.4


def test_classify_local_pref_fires_on_metric_divergence():
    """Warning 4/5 — cross-device legs with IDENTICAL prefix sets and IDENTICAL as_path
    but DIFFERENT metrics. With prefix sets aligned, score_leak's `only_one_side`
    contribution is zero — so score_leak collapses to a low value while
    score_local_pref dominates via the metric diff. This isolates BGP_LOCAL_PREF
    deterministically; the prior assert-IN form let ROUTE_LEAK silently win
    because differing prefixes pushed score_leak higher than score_local_pref.
    """
    from app.security.pathcompute.classify import classify
    from infracanvas.graph.models import NetworkPath, PathHop
    fwd = NetworkPath(id="fwd", direction="forward",
                      hops=[PathHop(node_id="dev-a", source_ip="10.0.0.1", dest_ip="10.0.1.1")])
    ret = NetworkPath(id="ret", direction="return",
                      hops=[PathHop(node_id="dev-b", source_ip="10.0.1.1", dest_ip="10.0.0.1")])
    # Same prefix set on both legs → no `only_one_side` contribution to score_leak.
    # Same as_path → score_leak's AS-diff contribution is zero.
    # Different metric → score_local_pref >= 0.5 (dominates).
    fwd_routes = [type("R", (), {"prefix": "10.0.0.0/24", "as_path": "65001 65002", "metric": 100})()]
    ret_routes = [type("R", (), {"prefix": "10.0.0.0/24", "as_path": "65001 65002", "metric": 300})()]
    cause, conf, ev = classify(fwd, ret, fwd_routes, ret_routes, nat_rules=[])
    assert cause == "BGP_LOCAL_PREF"
    assert ev["scores"]["BGP_LOCAL_PREF"] >= 0.4
    assert ev["scores"].get("ROUTE_LEAK", 0.0) <= 0.4
    assert conf >= 0.4
