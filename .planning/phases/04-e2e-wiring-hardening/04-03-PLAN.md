    **Edit 4 — Add a new useEffect for global keyboard shortcuts** (Cmd/Ctrl+\\, 1, 2) with the suppression rule from UI-SPEC §Interaction Contract — Keyboard Shortcuts. Copy this verbatim:
    ```tsx
    useEffect(() => {
      const handler = (e: KeyboardEvent) => {
        // Suppress when focus is in a form field (per UI-SPEC suppression rule)
        const target = e.target as HTMLElement | null;
        const tag = target?.tagName;
        if (
          tag === 'INPUT' ||
          tag === 'TEXTAREA' ||
          tag === 'SELECT' ||
          target?.isContentEditable === true
        ) {
          return;
        }

        // Cmd/Ctrl + \ toggles Canvas <-> FlowMap
        if ((e.metaKey || e.ctrlKey) && e.key === '\\') {
          e.preventDefault();
          const next = activeTab === 'canvas' ? 'flowmap' : 'canvas';
          if (next === 'flowmap' &amp;&amp; !hasFlowMap) return; // no-op when FlowMap is disabled
          setActiveTab(next);
          return;
        }

        // '1' jumps to Canvas
        if (e.key === '1' &amp;&amp; !e.metaKey &amp;&amp; !e.ctrlKey &amp;&amp; !e.altKey) {
          setActiveTab('canvas');
          return;
        }

        // '2' jumps to FlowMap (no-op if disabled)
        if (e.key === '2' &amp;&amp; !e.metaKey &amp;&amp; !e.ctrlKey &amp;&amp; !e.altKey) {
          if (!hasFlowMap) return;
          setActiveTab('flowmap');
          return;
        }
      };
      window.addEventListener('keydown', handler);
      return () => window.removeEventListener('keydown', handler);
    }, [activeTab, hasFlowMap, setActiveTab]);
    ```
