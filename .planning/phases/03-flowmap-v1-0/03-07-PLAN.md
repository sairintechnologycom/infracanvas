    **`viewer/src/components/flowmap/edges/PathEdge.tsx`** (net-new pattern per PATTERNS.md):
    ```tsx
    import { BaseEdge, getSmoothStepPath, MarkerType, type EdgeProps } from '@xyflow/react';

    interface PathEdgeData {
      direction?: 'forward' | 'return' | 'both';
      pathId?: string;
    }

    export function PathEdge(props: EdgeProps) {
      const { sourceX, sourceY, targetX, targetY, data } = props;
      const [edgePath] = getSmoothStepPath({ sourceX, sourceY, targetX, targetY });
      const direction = (data as PathEdgeData | undefined)?.direction ?? 'both';

      const forwardStyle = {
        stroke: '#3B82F6',
        strokeWidth: 1.75,
        fill: 'none',
        transform: 'translate(0, -3px)',
      } as const;
      const returnStyle = {
        stroke: '#F97316',
        strokeWidth: 1.75,
        fill: 'none',
        transform: 'translate(0, 3px)',
      } as const;

      const renderForward = direction === 'forward' || direction === 'both';
      const renderReturn = direction === 'return' || direction === 'both';

      return (
        <>
          {renderForward && (
            <BaseEdge
              id={`${props.id}-forward`}
              path={edgePath}
              style={forwardStyle}
              markerEnd={`url(#arrow-forward)`}
            />
          )}
          {renderReturn && (
            <BaseEdge
              id={`${props.id}-return`}
              path={edgePath}
              style={returnStyle}
              markerStart={`url(#arrow-return)`}
            />
          )}
        </>
      );
    }

    // Marker definitions injected by FlowMapCanvas inside <svg><defs>.
    // All marker paths are pure JSX <path> — React auto-escapes; no raw HTML interpolation.
    export const pathEdgeMarkerDefs = (
      <defs>
        <marker id="arrow-forward" markerWidth={10} markerHeight={10} refX={8} refY={5} orient="auto">
          <path d="M0,0 L0,10 L8,5 z" fill="#3B82F6" />
        </marker>
        <marker id="arrow-return" markerWidth={10} markerHeight={10} refX={2} refY={5} orient="auto-start-reverse">
          <path d="M0,5 L8,0 L8,10 z" fill="#F97316" />
        </marker>
      </defs>
    );
    ```

    NOTE on marker strategy: ReactFlow 12's `markerEnd` prop accepts either a MarkerType object OR a string reference to an existing SVG marker. The PathEdge uses SVG marker refs (flexible for two-color arrows) — the `pathEdgeMarkerDefs` export is consumed by FlowMapCanvas in Task 2 via a `<svg><defs>` injection. If the MarkerType object form `{type: MarkerType.ArrowClosed, color: '#3B82F6'}` works cleanly for per-edge colors in ReactFlow 12.6, the executor should prefer that and delete the custom marker defs — mirror the existing `EDGE_STYLES` config in `viewer/src/lib/colors.ts`. Either path satisfies the UI-SPEC visual contract.

    import ELK from 'elkjs/lib/elk.bundled.js';
    import type { Node as RFNode, Edge as RFEdge } from '@xyflow/react';
    import type { ResourceGraph, ResourceNode as RNData } from '../../../types';

    const elk = new ELK();

    const NODE_W = 200;
    const NODE_H = 80;

    const ELK_OPTIONS = {
      'elk.algorithm': 'layered',
      'elk.direction': 'RIGHT',
      'elk.spacing.nodeNode': '80',
      'elk.layered.spacing.nodeNodeBetweenLayers': '120',
    };

    export function pickReactFlowNodeType(type: string): string {
      if (type === 'aws_ec2_transit_gateway' || type === 'azurerm_virtual_hub' || type === 'azurerm_virtual_wan') {
        return 'cloudHub';
      }
      if (type.includes('firewall') || type === 'aws_network_firewall' || type === 'azurerm_firewall') {
        return 'firewall';
      }
      if (type === 'aws_router' /* future */) return 'router';
      // Fallback — use cloudHub for any cloud network resource without a specific node type
      if (type.startsWith('aws_') || type.startsWith('azurerm_')) return 'cloudHub';
      return 'cloudHub';
    }

    export async function layoutFlowMap(
      graph: ResourceGraph,
      networkNodes: RNData[],
    ): Promise<{ nodes: RFNode[]; edges: RFEdge[] }> {
      if (networkNodes.length === 0) {
        return { nodes: [], edges: [] };
      }

      // Build elkjs input
      const elkGraph = {
        id: 'root',
        layoutOptions: ELK_OPTIONS,
        children: networkNodes.map((n) => ({ id: n.id, width: NODE_W, height: NODE_H })),
        edges: graph.edges
          .filter((e) => networkNodes.some((n) => n.id === e.source) && networkNodes.some((n) => n.id === e.target))
          .map((e) => ({ id: `${e.source}->${e.target}`, sources: [e.source], targets: [e.target] })),
      };

      const layouted = await elk.layout(elkGraph);

      const rfNodes: RFNode[] = (layouted.children ?? []).map((c) => {
        const data = networkNodes.find((n) => n.id === c.id)!;
        return {
          id: c.id,
          type: pickReactFlowNodeType(data.type),
          position: { x: c.x ?? 0, y: c.y ?? 0 },
          data,
        };
      });

      const rfEdges: RFEdge[] = (layouted.edges ?? []).map((e) => ({
        id: e.id,
        source: (e.sources ?? [])[0] ?? '',
        target: (e.targets ?? [])[0] ?? '',
        type: 'path',
        data: { direction: 'both' },
      }));

      return { nodes: rfNodes, edges: rfEdges };
    }
